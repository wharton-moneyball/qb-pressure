4. 核心函数：计算 PRSS（Pocket Rate of Space Shrinkage）

relevant_positions <- c("QB", "OL", "DL", "LB", "TE", "CB", "S")

# k: rolling window size
compute_qb_area <- function(play_data, k) {

  # Find the snap frame and the pass/ sack / spike frame
  snap_vals    <- play_data$frameId[play_data$event == "ball_snap"]
  release_vals <- play_data$frameId[play_data$event %in%
                    c("pass_forward", "qb_sack", "qb_spike")]

  if (length(snap_vals) == 0 || length(release_vals) == 0) {
    return(tibble(qb = NA_character_, max_neg_change = NA_real_))
  }

  snap_frame    <- min(snap_vals,    na.rm = TRUE)
  release_frame <- min(release_vals, na.rm = TRUE)

  if (!is.finite(snap_frame) || !is.finite(release_frame)) {
    return(tibble(qb = NA_character_, max_neg_change = NA_real_))
  }

  play_data <- play_data %>%
    filter(frameId >= snap_frame, frameId <= release_frame)

  # QB name
  qb_name <- play_data %>%
    filter(officialPosition == "QB") %>%
    distinct(displayName) %>%
    pull() %>%
    paste(collapse = "/")

  frames <- unique(play_data$frameId)

  # 需要至少 k+1 帧才能计算滚动均值
  if (length(frames) < (k + 1)) {
    return(tibble(qb = qb_name, max_neg_change = NA_real_))
  }

  # 逐帧计算QB的Voronoi面积
  qb_areas <- map_dfr(frames, function(f) {
    frame_data <- play_data %>%
      filter(
        frameId == f,
        !is.na(x), !is.na(y),
        officialPosition %in% relevant_positions
      ) %>%
      distinct(x, y, .keep_all = TRUE)

    if (nrow(frame_data) < 3) return(tibble(frameId = f, qb_area = NA_real_))

    tryCatch({
      vor    <- deldir(frame_data$x, frame_data$y)
      vor_df <- vor$summary %>%
        select(x, y, dir.area) %>%
        left_join(frame_data, by = c("x", "y"))
      qb_area <- vor_df %>%
        filter(officialPosition == "QB") %>%
        pull(dir.area)
      tibble(frameId = f, qb_area = ifelse(length(qb_area) > 0, qb_area[1], NA_real_))
    }, error = function(e) tibble(frameId = f, qb_area = NA_real_))
  }) %>%
    arrange(frameId) %>%
    mutate(
      rate_change = c(NA, diff(qb_area)),
      rolling_avg = suppressWarnings(
        zoo::rollmean(rate_change, k = k, align = "left", fill = NA)
      )
    )

  max_neg <- suppressWarnings(min(qb_areas$rolling_avg, na.rm = TRUE))
  if (!is.finite(max_neg)) max_neg <- NA_real_

  tibble(qb = qb_name, max_neg_change = max_neg)
}
